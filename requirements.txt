# Portfolio Project - Installation Requirements
# Next.js Portfolio with 3D Earth Background and Certificate Management

# Prerequisites
# Node.js >= 18.0.0
# npm >= 8.0.0 or yarn >= 1.22.0

# Installation Steps:
# 1. Clone the repository
# 2. Run: npm install
# 3. Copy .env.example to .env.local and set your environment variables
# 4. Run: npm run dev

# Core Dependencies
next@^14.0.0
react@^18.0.0
react-dom@^18.0.0

# UI and Animation Libraries
framer-motion@^10.0.0
tailwindcss@^3.0.0

# 3D Graphics
three@^0.160.0

# Email Integration
@emailjs/browser@^4.4.1

# Development Dependencies
typescript@^5.0.0
@types/node@^20.0.0
@types/react@^18.0.0
@types/three@^0.179.0
autoprefixer@^10.0.0
postcss@^8.0.0

# Global Dependencies (install with -g flag)
# vercel@latest (for deployment)

# Environment Variables Required:
# NEXT_PUBLIC_ADMIN_PASSWORD=your_secure_password
# NEXT_PUBLIC_EMAILJS_SERVICE_ID=your_service_id (optional)
# NEXT_PUBLIC_EMAILJS_TEMPLATE_ID=your_template_id (optional)
# NEXT_PUBLIC_EMAILJS_PUBLIC_KEY=your_public_key (optional)

# Quick Installation Commands:
# npm install
# npm install -g vercel (for deployment)

# Build Commands:
# npm run dev        # Development server
# npm run build      # Production build
# npm run start      # Production server
# npm run lint       # Code linting

# Deployment:
# vercel             # Deploy to Vercel
